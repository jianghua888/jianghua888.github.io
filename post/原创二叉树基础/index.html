【原创】二叉树基础

2020年5月17日 
#include<stdio.h>
#include<stdlib.h>
//定义数据节点 
typedef struct Node {
	char ch;
	struct Node *leftChildPtr;
	struct Node *rightChildPtr;
}TreeNode,*TreeNodePtr;

char *str = "abc##d##ef##g##";  //要创建的节点数据

/**
 * 创建一课二叉树 
 */ 
TreeNodePtr createBinaryTree(char ch) 
{
	if(ch == '\0' || ch == '#') {
		return NULL;
	}
	//如果不是#,则创建一个节点
	TreeNodePtr ptr = (TreeNodePtr)malloc(sizeof(TreeNode));
	ptr->ch = ch;
	str++; //外面指针走一个字符 
	ptr->leftChildPtr = createBinaryTree(*str);	
	str++; 
	ptr->rightChildPtr = createBinaryTree(*str);
	return ptr;
} 

/**
 * 前序遍历 
 */ 
void preTraverseTree(TreeNodePtr node)
{
	if(node != NULL) 
	{
		//std::cout << node->ch << " -- ";
		printf("%c",node->ch);
		printf(" -- ");
		preTraverseTree(node->leftChildPtr);
		preTraverseTree(node->rightChildPtr);
	}	
}

/**
 * 中序遍历 
 */
void inOrderTraverseTree(TreeNodePtr node)
{
	if(node != NULL) 
	{
		inOrderTraverseTree(node->leftChildPtr);
		printf("%c",node->ch);
		printf(" -- ");
		inOrderTraverseTree(node->rightChildPtr);
	}	
}

/**
 * 后续遍历 
 */
void postOrderTraverseTree(TreeNodePtr node)
{
	if(node != NULL) 
	{
		postOrderTraverseTree(node->leftChildPtr);
		postOrderTraverseTree(node->rightChildPtr);
		printf("%c",node->ch);
		printf(" -- ");
	}	
}

int main() {
	
	char ch = *str;
	TreeNodePtr root = createBinaryTree(ch);
    
	createBinaryTree(root);
	printf("前序遍历:\n"); 
	preTraverseTree(root); 
	printf("\n"); 
	
	printf("中序遍历:\n");
	inOrderTraverseTree(root); 
	printf("\n"); 
	
	printf("后序遍历:\n");
	postOrderTraverseTree(root); 
	printf("\n"); 
	return 0;
}

执行结果

前序遍历: a – b – c – d – e – f – g –
中序遍历: c – b – d – a – f – e – g –
后序遍历: c – d – b – f – g – e – a –
